#### 3. file.service.ts 核心方法

```typescript
@Injectable()
export class FileService {
  constructor(
    private prisma: PrismaService,
    private configService: ConfigService,
  ) {}

  async uploadFile(
    userId: number,
    file: Express.Multer.File,
    fileType: 'image' | 'video' | 'document' | 'other' = 'image',
  ) {
    // 文件验证
    this.validateFile(file, fileType);

    // 生成文件名
    const fileName = this.generateFileName(file.originalname);

    // 保存文件到存储
    const filePath = await this.saveFile(file, fileName);

    // 保存文件记录
    const fileRecord = await this.prisma.file.create({
      data: {
        userId,
        fileName,
        originalName: file.originalname,
        fileSize: file.size,
        mimeType: file.mimetype,
        filePath,
        fileType,
        isActive: true,
      },
    });

    return {
      id: fileRecord.id,
      fileName: fileRecord.originalName,
      url: this.getFileUrl(fileRecord.filePath),
      fileSize: fileRecord.fileSize,
    };
  }

  async findAllByUserId(userId: number) {
    return this.prisma.file.findMany({
      where: { userId, isActive: true },
      orderBy: { createdAt: 'desc' },
    });
  }

  async findOne(id: number) {
    const file = await this.prisma.file.findUnique({
      where: { id },
    });

    if (!file) {
      throw new NotFoundException('文件不存在');
    }

    return file;
  }

  async remove(userId: number, fileId: number) {
    const file = await this.prisma.file.findUnique({
      where: { id: fileId },
    });

    if (!file) {
      throw new NotFoundException('文件不存在');
    }

    if (file.userId !== userId) {
      throw new ForbiddenException('无权删除此文件');
    }

    await this.prisma.file.update({
      where: { id: fileId },
      data: { isActive: false },
    });

    // 异步删除物理文件
    this.deletePhysicalFile(file.filePath);

    return { message: '文件已删除' };
  }

  private validateFile(file: Express.Multer.File, fileType: string) {
    // 文件大小验证
    const maxSizes = {
      image: 10 * 1024 * 1024, // 10MB
      video: 100 * 1024 * 1024, // 100MB
      document: 50 * 1024 * 1024, // 50MB
      other: 100 * 1024 * 1024, // 100MB
    };

    if (file.size > maxSizes[fileType]) {
      throw new BadRequestException(`文件大小不能超过 ${maxSizes[fileType] / 1024 / 1024}MB`);
    }

    // 文件类型验证
    const allowedMimeTypes = {
      image: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
      video: ['video/mp4', 'video/webm', 'video/avi'],
      document: ['application/pdf', 'application/msword', 'text/plain'],
      other: ['application/octet-stream'],
    };

    if (!allowedMimeTypes[fileType].includes(file.mimetype)) {
      throw new BadRequestException('不支持的文件类型');
    }
  }

  private generateFileName(originalName: string) {
    const ext = extname(originalName);
    const name = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    return `${name}${ext}`;
  }

  private async saveFile(file: Express.Multer.File, fileName: string) {
    const uploadDir = this.configService.get('UPLOAD_DIR', './uploads');
    const datePath = new Date().toISOString().split('T')[0];
    const dirPath = path.join(uploadDir, datePath);

    if (!fs.existsSync(dirPath)) {
      fs.mkdirSync(dirPath, { recursive: true });
    }

    const filePath = path.join(datePath, fileName);
    await fs.promises.writeFile(path.join(uploadDir, filePath), file.buffer);

    return filePath;
  }

  private getFileUrl(filePath: string) {
    const baseUrl = this.configService.get('BASE_URL', 'http://localhost:3000');
    return `${baseUrl}/uploads/${filePath}`;
  }

  private deletePhysicalFile(filePath: string) {
    const uploadDir = this.configService.get('UPLOAD_DIR', './uploads');
    const fullPath = path.join(uploadDir, filePath);

    fs.promises.unlink(fullPath).catch(err => {
      this.logger.error('删除文件失败', err.stack);
    });
  }
}
```

### 接口 1: 上传文件

| 字段 | 内容 |
|------|------|
| **地址** | `POST /api/v1/files/upload` |
| **认证** | Bearer Token |
| **请求格式** | `multipart/form-data` |

**请求参数**:
```
file: 要上传的文件
fileType: 可选，文件类型（image, video, document, other），默认 image
```

```json
{
  "code": 200,
```
