```typescript
export class SearchQueryDto {
  @IsString()
  @ApiProperty({ description: '搜索关键词' })
  q: string;

  @IsString()
  @IsOptional()
  @ApiPropertyOptional({ description: '搜索类型' })
  type?: 'user' | 'post' | 'comment' | 'tag' | 'all';

  @IsNumber()
  @IsOptional()
  @ApiPropertyOptional({ description: '页码' })
  page?: number = 1;

  @IsNumber()
  @IsOptional()
  @ApiPropertyOptional({ description: '每页数量' })
  limit?: number = 10;

  @IsString()
  @IsOptional()
  @ApiPropertyOptional({ description: '排序方式' })
  sort?: 'relevance' | 'date' | 'popularity';
}
```

#### 2. search-result.dto.ts

```typescript
export class SearchResultDto {
  @ApiProperty({ description: '结果类型' })
  type: 'user' | 'post' | 'comment' | 'tag';

  @ApiProperty({ description: '结果数据' })
  data: any;

  @ApiProperty({ description: '相关度分数' })
  score: number;
}
```

#### 3. search.service.ts 核心方法

```typescript
@Injectable()
export class SearchService {
  constructor(private prisma: PrismaService) {}

  async search(query: SearchQueryDto) {
    const { q, type = 'all', page = 1, limit = 10, sort = 'relevance' } = query;

    const searchResults: SearchResultDto[] = [];

    // 搜索用户
    if (type === 'all' || type === 'user') {
      const users = await this.prisma.user.findMany({
        where: {
          OR: [
            { username: { contains: q, mode: 'insensitive' } },
            { nickname: { contains: q, mode: 'insensitive' } },
          ],
          isActive: true,
        },
        take: limit,
        skip: (page - 1) * limit,
      });

      searchResults.push(
        ...users.map(user => ({
          type: 'user',
          data: { id: user.id, username: user.username, nickname: user.nickname },
          score: 0.8,
        })),
      );
    }

    // 搜索帖子
    if (type === 'all' || type === 'post') {
      const posts = await this.prisma.post.findMany({
        where: {
          OR: [
            { title: { contains: q, mode: 'insensitive' } },
            { content: { contains: q, mode: 'insensitive' } },
          ],
          isActive: true,
        },
        take: limit,
        skip: (page - 1) * limit,
      });

      searchResults.push(
        ...posts.map(post => ({
          type: 'post',
          data: { id: post.id, title: post.title, content: post.content },
          score: 0.9,
        })),
      );
    }

    // 搜索评论
    if (type === 'all' || type === 'comment') {
      const comments = await this.prisma.comment.findMany({
        where: { content: { contains: q, mode: 'insensitive' }, isActive: true },
        take: limit,
        skip: (page - 1) * limit,
      });

      searchResults.push(
        ...comments.map(comment => ({
          type: 'comment',
          data: { id: comment.id, content: comment.content },
          score: 0.7,
        })),
      );
    }

    // 搜索标签
    if (type === 'all' || type === 'tag') {
      const tags = await this.prisma.tag.findMany({
        where: { name: { contains: q, mode: 'insensitive' }, isActive: true },
        take: limit,
        skip: (page - 1) * limit,
      });

      searchResults.push(
        ...tags.map(tag => ({
          type: 'tag',
          data: { id: tag.id, name: tag.name },
          score: 0.85,
        })),
      );
    }

    // 结果排序
    this.sortResults(searchResults, sort);

    // 分页
    const start = (page - 1) * limit;
    const end = start + limit;
    const paginatedResults = searchResults.slice(start, end);

    return {
      total: searchResults.length,
      page,
      limit,
      results: paginatedResults,
    };
  }

  private sortResults(results: SearchResultDto[], sort: string) {
    switch (sort) {
      case 'relevance':
        results.sort((a, b) => b.score - a.score);
        break;
      case 'date':
        results.sort((a, b) => new Date(b.data.createdAt).getTime() - new Date(a.data.createdAt).getTime());
        break;
      case 'popularity':
        results.sort((a, b) => (b.data.likeCount || 0) - (a.data.likeCount || 0));
        break;
    }
  }
}
```

## 接口列表

### 接口 1: 执行搜索

| 字段 | 内容 |
|------|------|
| **地址** | `GET /api/v1/search` |
| **认证** | Bearer Token |
| **请求格式** | - |

**查询参数**:
```
q: 搜索关键词
type: 搜索类型（all/user/post/comment/tag）
page: 页码（默认 1）
limit: 每页数量（默认 10）
sort: 排序方式（relevance/date/popularity）
```

**成功响应**:
```json
{
  "code": 200,
  "message": "success",
  "data": {
    "total": 15,
    "page": 1,
    "limit": 10,

        "type": "user",
        "data": {
          "id": 1,
          "username": "testuser",
          "nickname": "测试用户"

        "score": 0.8
      }
    ]
  },
  "timestamp": "2026-04-24T10:30:00.000Z"
}
```

## 风险评估

### 技术风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 搜索性能 | 搜索速度慢 | 中 | 使用全文搜索引擎和缓存 |
| 搜索精度 | 搜索结果不相关 | 中 | 优化搜索算法 |

### 业务风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 搜索滥用 | 恶意搜索导致服务器负载 | 低 | 搜索限流和限制 |
| 搜索内容 | 搜索到不恰当的内容 | 低 | 内容审核机制 |
