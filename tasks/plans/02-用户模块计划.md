#### 3. user.service.ts 核心方法

```typescript
@Injectable()
export class UserService {
  constructor(private prisma: PrismaService) {}

  async findProfile(userId: number) {
    const user = await this.prisma.user.findUnique({
      where: { id: userId },
      select: {
        id: true,
        username: true,
        email: true,
        avatarUrl: true,
        nickname: true,
        bio: true,
        gender: true,
        birthday: true,
        city: true,
        createdAt: true,
      },
    });
    
    if (!user) {
      throw new NotFoundException('用户不存在');
    }
    
    return user;
  }

  async update(userId: number, updateUserDto: UpdateUserDto) {
    // 检查用户名重复
    if (updateUserDto.username) {
      const existingUser = await this.prisma.user.findFirst({
        where: { username: updateUserDto.username, NOT: { id: userId } },
      });
      
      if (existingUser) {
        throw new ConflictException('用户名已被使用');
      }
    }

    return this.prisma.user.update({
      where: { id: userId },
      data: updateUserDto,
    });
  }

  async changePassword(userId: number, changePasswordDto: ChangePasswordDto) {
    const { oldPassword, newPassword } = changePasswordDto;
    
    const user = await this.prisma.user.findUnique({
      where: { id: userId },
    });
    
    if (!user) {
      throw new NotFoundException('用户不存在');
    }

    const isPasswordValid = await bcrypt.compare(oldPassword, user.passwordHash);
    
    if (!isPasswordValid) {
      throw new UnauthorizedException('旧密码错误');
    }

    const passwordHash = await bcrypt.hash(newPassword, 10);
    
    return this.prisma.user.update({
      where: { id: userId },
      data: { passwordHash },
    });
  }

  async uploadAvatar(userId: number, avatarUrl: string) {
    return this.prisma.user.update({
      where: { id: userId },
      data: { avatarUrl },
    });
  }
}
```

## 风险评估

### 技术风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 头像文件过大 | 上传时间过长 | 中 | 文件大小限制和压缩 |
| 并发修改 | 数据不一致 | 低 | 事务和版本控制 |

### 业务风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 个人信息泄露 | 用户数据暴露 | 低 | HTTPS 和加密传输 |
| 大量用户同时修改 | 服务压力 | 低 | 监控和限流 |
